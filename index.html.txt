  function parseText(t){
    const r={};

    const fcp=t.match(/Fecha\s*:\s*Guayaquil,\s*(\d{2}\/\d{2}\/\d{4})/i) || t.match(RGX.fecha); if(fcp) r.f_cpago=fcp[1];

    const sb=t.match(RGX.subtotal15); if(sb) r.base=toNum(sb[1]);
    const iv=t.match(RGX.iva15); if(iv) r.iva=toNum(iv[1]);
    const tt=t.match(RGX.total); if(tt) r.total=toNum(tt[1]);
    const rn=t.match(RGX.retNum); if(rn) r.n_ret=rn[1];
    let ret=[]; if(RGX.retenIva100.test(t)) ret.push('IVA 100%'); const rr=t.match(RGX.retenRentaPct); if(rr) ret.push(`RENTA ${rr[1]}%`);

  function finalizeRec(rec){
    if(!rec.ruc){ const rr=(rec.__allText||'').match(/\bRUC\s*\.?:\s*(\d{13})/i); if(rr) rec.ruc=rr[1]; }
    if(!rec.nombre && rec.ruc){
      const lines=(rec.__allText||'').split(/\n/);
      for(let i=0;i<lines.length;i++){ if(lines[i].includes(rec.ruc) && lines[i+1]){ rec.nombre=(rec.nombre||lines[i+1].trim()); break; } }
    }
    let rI=rec.ret_iva_val, rR=rec.ret_renta_val;
    if(rI==null && rec.ret_desc && /IVA\s*100%/i.test(rec.ret_desc) && rec.iva!=null) rI=rec.iva;
    if(rR==null && rec.ret_desc && /RENTA\s*([\d\.]+)%/i.test(rec.ret_desc) && rec.base!=null){
      const pct=parseFloat(rec.ret_desc.match(/RENTA\s*([\d\.]+)%/i)[1]||'0'); rR=+(rec.base*(pct/100)).toFixed(2);
    }
    if(rec.total!=null && (rI!=null || rR!=null)) rec.pagar=+(rec.total-((rI||0)+(rR||0))).toFixed(2);
    if(!rec.tipo) rec.tipo='GEN';
    return rec;
